CREATE DATABASE socialnet;
use socialnet;
CREATE TABLE users (
    id int PRIMARY KEY AUTO_INCREMENT,
    username varchar(250),
    passwords varchar(250),
    firstname varchar(250),
    lastname varchar(250),
    phone varchar(13),
    age int,
    email varchar(250),
    gender boolean,
    avata varchar(250),
    type varchar(15)
);
